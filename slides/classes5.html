class Foo {
      constructor(bar) {
        this.bar = bar;
      }
    }

    class FooBar extends Foo {
      constructor(bar) {
        super(bar);
      }
    }

    let fooBar = new FooBar('baz');
    console.log(fooBar.bar); // baz
    fooBar instanceof Foo // true
