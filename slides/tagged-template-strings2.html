let name = 'Dave';

    function upper(strs, ...vals) {
      return strs.reduce(function (p, c, i) {
        if (i > 0) {
          p += vals[i-1].toUpperCase();
        }
        return p + c;
      }, '');
    }

    let greet = upper`Good morning ${name}`;
    console.log(greet); // Good morning DAVE
