function upper(strs, ...vals) {
      return strs.reduce(function (p, c, i) {
        if (i > 0) {
          p += vals[i-1].toUpperCase();
        }
        return p + c;
      }, '');
    }

    let him = 'Dave', me = 'Hal';
    let str = upper`Hello ${him}, it's ${me}`;
    console.log(str); // Hello DAVE, it's HAL
